#! the aggregation will be run within the scope of the match (country) query
GET /avatar/_search
{
  "query": {
    "match": {
      "country": "France"
    }
  }, 
    "aggs" : {
        "main_interests" : {
            "terms" : { "field" : "interests.keyword" },
            "aggs" : {
                "avg_age" : {
                    "avg" : { "field" : "age" }
                },
                "max_age":{
                  "max":{
                    "field": "age"
                  }
                }
            }
        }
    }
}

                "my_stopwords": {
                    "type":       "stop",

                "my_analyzer": {
                    "type":         "custom",
